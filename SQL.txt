В psql названия таблиц и полей с заглавными буквами надо брать в ковычки.

Задание 1

SELECT login, COUNT("Orders".*)
FROM "Couriers"
JOIN "Orders" ON "Couriers".id="Orders"."courierId"
WHERE "Orders"."inDelivery"=true GROUP BY "Couriers".login;

Задание 2
SELECT track,
CASE
  WHEN "inDelivery"=true THEN 1
  WHEN finished=true THEN 2
  WHEN canсelled=true THEN -1
  ELSE 0
END AS status
FROM "Orders";
